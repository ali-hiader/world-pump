function ResultsSummary({ total, page, limit }: { total: number; page: number; limit: number }) {
  if (total === 0) return null;
  const start = (page - 1) * limit + 1;
  const end = Math.min(total, page * limit);
  return (
    <p className="text-sm text-muted-foreground">Showing {start}â€“{end} of {total} products</p>
  );
}
